39 class ReadMostlySharedPtrCore {
40  public:
41  T* get() {
42  return ptrRaw_;
43  }
44 
45  std::shared_ptr<T> getShared() {
46  return ptr_;
47  }
48 
49  bool incref() {
50  return ++count_ > 0;
51  }
52 
53  void decref() {
54  if (--count_ == 0) {
55  ptrRaw_ = nullptr;
56  ptr_.reset();
57 
58  decrefWeak();
59  }
60  }
61 
62  void increfWeak() {
63  auto value = ++weakCount_;
64  DCHECK_GT(value, 0);
65  }
66 
67  void decrefWeak() {
68  if (--weakCount_ == 0) {
69  delete this;
70  }
71  }
72 
73  size_t useCount() const {

75  }

78  assert(*count_ == 0);
79  assert(*weakCount_ == 0);
80  }
