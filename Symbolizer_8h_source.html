86 template <size_t N>
87 bool fixFrameArray(FrameArray<N>& fa, ssize_t n) {
88  if (n != -1) {
89  fa.frameCount = n;
90  for (size_t i = 0; i < fa.frameCount; ++i) {
91  fa.frames[i].found = false;
92  }
93  return true;
94  } else {
95  fa.frameCount = 0;
96  return false;
97  }
98 }
