41 bool ThreadedRepeatingFunctionRunner::stopImpl() {
42  {
43  std::unique_lock<std::mutex> lock(stopMutex_);
44  if (stopping_) {
45  return false; // Do nothing if stop() is called twice.
46  }
47  stopping_ = true;
48  }
49  stopCv_.notify_all();
50  for (auto& t : threads_) {
51  t.join();
52  }
53  return true;
54 }
