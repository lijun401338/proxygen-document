34 void runInThreadsAndWait(size_t nthreads, function<void(size_t)> cb) {
35  vector<thread> threads(nthreads);
36  for (size_t i = 0; i < nthreads; ++i) {
37  threads[i] = thread(cb, i);
38  }
39  for (size_t i = 0; i < nthreads; ++i) {
40  threads[i].join();
41  }
42 }
