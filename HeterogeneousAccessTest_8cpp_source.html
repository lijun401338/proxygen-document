149

152  EXPECT_NE(hash(lhs2), hash(rhs1)); // technically only low probability
153  EXPECT_EQ(hash(lhs2), hash(rhs2));

159 
160  smaller.resize(1);
161  EXPECT_FALSE(equalTo(a, lhs1));
