109 inline size_t encodeVarint(uint64_t val, uint8_t* buf) {
110  uint8_t* p = buf;
111  while (val >= 128) {
112  *p++ = 0x80 | (val & 0x7f);
113  val >>= 7;
114  }
115  *p++ = uint8_t(val);
116  return size_t(p - buf);
117 }
