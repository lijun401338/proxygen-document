28 static int wrapPositional(F f, int fd, off_t offset, Args... args) {
29  off_t origLoc = lseek(fd, 0, SEEK_CUR);
30  if (origLoc == off_t(-1)) {
31  return -1;
32  }
33  if (lseek(fd, offset, SEEK_SET) == off_t(-1)) {
34  return -1;
35  }
36 
37  int res = (int)f(fd, args...);
38 
39  int curErrNo = errno;

41  if (res == -1) {
42  errno = curErrNo;
43  }
44  return -1;
45  }
46  errno = curErrNo;

49 }
