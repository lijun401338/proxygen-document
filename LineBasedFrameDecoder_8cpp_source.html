94 int64_t LineBasedFrameDecoder::findEndOfLine(IOBufQueue& buf) {
95  Cursor c(buf.front());
96  for (uint32_t i = 0; i < maxLength_ && i < buf.chainLength(); i++) {
97  auto b = c.read<char>();
98  if (b == '\n' && terminatorType_ != TerminatorType::CARRIAGENEWLINE) {
99  return i;
100  } else if (terminatorType_ != TerminatorType::NEWLINE &&
101  b == '\r' && !c.isAtEnd() && c.read<char>() == '\n') {
102  return i;
103  }
104  }
105
