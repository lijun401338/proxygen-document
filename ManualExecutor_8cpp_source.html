35 size_t ManualExecutor::run() {

37  size_t n;
38  Func func;
39 
40  {

42 
43  while (!scheduledFuncs_.empty()) {
44  auto& sf = scheduledFuncs_.top();
45  if (sf.time > now_) {
46  break;
47  }
48  funcs_.emplace(sf.moveOutFunc());

50  }
51

54
