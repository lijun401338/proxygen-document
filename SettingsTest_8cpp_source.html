50 struct UserDefinedType {
51  explicit UserDefinedType(folly::StringPiece value) {
52  if (value == "a") {
53  value_ = 0;
54  } else if (value == "b") {
55  value_ = 100;
56  } else {
57  throw std::runtime_error("Invalid value passed to UserDefinedType ctor");
58  }
59  }
60 
61  bool operator==(const UserDefinedType& other) const {
62  return value_ == other.value_;
63  }
64 
65  int value_;
66 };
