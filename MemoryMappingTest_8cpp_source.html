87 void writeStringToFileOrDie(const std::string& str, int fd) {
88  const char* b = str.c_str();
89  size_t count = str.size();
90  ssize_t total_bytes = 0;
91  ssize_t r;
92  do {
93  r = write(fd, b, count);
94  if (r == -1) {
95  if (errno == EINTR) {
96  continue;
97  }
98  PCHECK(r) << "write";
99  }
100 
101  total_bytes += r;
102  b += r;
103  count -= r;
104  } while (r != 0 && count);
105 }
